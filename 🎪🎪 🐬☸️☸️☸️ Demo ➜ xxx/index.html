🔵 k8s - yml

🔶 create namespace: ns-tool-net 🔶 create pod yaml: toolNetNettoolsPod.yaml

apiVersion: v3
kind: Pod
metadata:
name: toolNetNettoolsPod
namespace: ns-tool-net
spec:
containers:
- name: toolNetNettoolsPod
    image: travelping/nettools:latest
    command:
    - sleep
    - "infinity"
    imagePullPolicy: IfNotPresent
restartPolicy: Never
